COMANDOS MySQL:

CREATE DATABASE Projeto1
CREATE TABLE Cadastro (id INT NOT NULL AUTO_INCREMENT, usuario VARCHAR(20) NOT NULL, senha VARCHAR(10), PRIMARY KEY(id));
CREATE TABLE Post (id INT NOT NULL AUTO_INCREMENT, titulo VARCHAR(50) NOT NULL, data DATE, usuario INT NOT NULL REFERENCES Cadastro(id), texto TEXT, PRIMARY KEY(id));
SET GLOBAL time_zone = '-3:00'; 
CREATE TABLE Login (id_usuario INT NOT NULL AUTO_INCREMENT, usuario VARCHAR(20) NOT NULL, senha VARCHAR(10), PRIMARY KEY (id_usuario));
